\usepackage[sort = def, nonumberlist]{glossaries}
\renewcommand*{\glstextformat}[1]{\textcolor{black}{#1}}
\newglossary{symbols}{sym}{sbl}{List of Abbreviations and Symbols}
\makeglossary



% data stuff
\newglossaryentry{X}{type=symbols,name={\ensuremath{\X}},
	description={a design matrix, usually with dimensions $N \times p$}}

\newglossaryentry{y}{type=symbols,name={\ensuremath{\y}},
	description={a response vector of the binary classes}}

\newglossaryentry{U}{type=symbols,name={\ensuremath{\U}},
	description={of the svd of $\X = \U \D \V^T$}}

\newglossaryentry{D}{type=symbols,name={\ensuremath{\D}},
	description={of the svd of $\X = \U \D \V^T$}}

\newglossaryentry{V}{type=symbols,name={\ensuremath{\V}},
	description={of the svd of $\X = \U \D \V^T$}}

\newglossaryentry{W}{type=symbols,name={\ensuremath{\W}},
	description={a weight matrix}}

\newglossaryentry{pseudoX}{type=symbols,name={\ensuremath{\widetilde{\X}}},
	description={a weighted design matrix}}

\newglossaryentry{z}{type=symbols,name={\ensuremath{\z}},
	description={a psuedo response based on the weight $\W$}}

\newglossaryentry{prob}{type=symbols,name={\ensuremath{\p}},
	description={a vector of probabilities of data belonging to class 1}}

\newglossaryentry{probhat}{type=symbols,name={\ensuremath{\widehat{\p}}},
	description={a fitted vector of probabilities of data belonging to class 1, typically superscripted with the method of fit}}

% coef stuff

\newglossaryentry{beta}{type=symbols,name={\ensuremath{\bbeta}},
	description={a coefficient vector in the original coordinates}}

\newglossaryentry{betahat}{type=symbols,name={\ensuremath{\widehat{\bbeta}}},
	description={a fitted coefficient vector in the original coordinates, typically superscripted with the method of fit}}

\newglossaryentry{gamma}{type=symbols,name={\ensuremath{\bgamma}},
	description={a coefficient vector in the principal components space}}

\newglossaryentry{gammahat}{type=symbols,name={\ensuremath{\widehat{\bgamma}}},
	description={a fitted coefficient vector in the principal components space, typically superscripted with the method of fit}}

% misc

\newglossaryentry{penalty}{type=symbols,name={\ensuremath{p_{\lambda}}},
	description={a penalty function with parameter $\lambda$}}

\newglossaryentry{loglike}{type=symbols,name={\ensuremath{\ell}},
	description={the log likelihood function}}
